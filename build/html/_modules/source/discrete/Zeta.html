[docs]    def pmf(self, x: List[int] = None) -> Union[int, float, List[int]]:
        """
        Args:

            x (List[int]): random variable or list of random variables

        Returns: 
            either probability mass evaluation for some point or scatter plot of Zeta distribution.
        """
        s = self.s
        def __generator(s, k): return (1 / k**s) / _zeta(s)

        if x is not None and isinstance(x, List):
            return [__generator(s, i) for i in x]  # double check this function

        return __generator(s, x)

        """
        Returns: 
            the variance of Zeta Distribution. Returns None if undefined.
        """
        s = self.s
        if s > 3:
            _x0 = _zeta(s)
            return (_zeta(s-2)/_x0) - (_zeta(s-1)/_zeta(s))**2
        return "undefined"

[docs]    def skewness(self) -> Union[str, float]:
        """
        Returns: 
            the skewness of Zeta Distribution.
        """
        s = self.s
        if s <= 4:
            return "undefined"
        _x0 = _zeta(s-2)*_zeta(s)
        _x1 = _zeta(s-1)
        return (_zeta(s-3)*_zeta(s)**2 - 3*_x1*_x0 + 2*_x1**3) / pow(_x0-_x1**2, 3/2)



[docs]    def kurtosis(self) -> Union[str, float]:
        """
        Returns: 
            the kurtosis of Zeta Distribution.
        """
        s = self.s 
        if s <= 5:
            return "undefined"

        _x0 = _zeta(s-2)
        _x1 = _zeta(s)
        _x3 = _zeta(s-1)

        scale = 1/pow(_x0*_x1 - _x3**2,2)
        numerator = (_zeta(s-4)*_x1**3) - (4*_x3*_zeta(s-3)*_x1**2) + (6*_x3**2*_x0*_x1-3*_x3**4)
        return scale*numerator
