[docs]    def mean(self) -> float:
        """
        Returns: 
            the mean of Binomial Distribution.
        """
        return self.n * self.p



[docs]    def median(self) -> Tuple[int, int]:
        """
        Returns: 
            the median of Binomial Distribution. Either one defined in the tuple of result.
        """
        n = self.n
        p = self.p
        return _floor(n * p), _ceil(n * p)



[docs]    def mode(self) -> Tuple[int, int]:
        """
        Returns: 
            the mode of Binomial Distribution. Either one defined in the tuple of result.
        """
        n = self.n
        p = self.p
        return _floor((n + 1) * p), _ceil((n + 1) * p) - 1



[docs]    def var(self) -> float:
        """
        Returns: 
            the variance of Binomial Distribution.
        """
        n = self.n
        p = self.p
        q = 1 - p
        return n * p * q



[docs]    def skewness(self) -> float:
        """
        Returns: 
            the skewness of Binomial Distribution.
        """
        n = self.n
        p = self.p
        q = 1 - p
        return (q - p) / _sqrt(n * p * q)



[docs]    def kurtosis(self) -> float:
        """
        Returns: 
            the kurtosis of Binomial Distribution.
        """
        n = self.n
        p = self.p
        q = 1 - p
        return (1 - 6 * p * q) / (n * p * q)



[docs]    def keys(self) -> Dict[str, Union[float, int]]:
        """
        Summary statistic regarding the Binomial distribution which contains the following parts of the distribution:
        (mean, median, mode, var, std, skewness, kurtosis).

        Returns:
            Dict[str, Union[float, int]]: [description]
        """
        return {
            'mean': self.mean(), 'median': self.median(), 'mode': self.mode(),
            'var': self.var(), 'std': self.std(), 'skewness': self.skewness(), 'kurtosis': self.kurtosis()
        }
